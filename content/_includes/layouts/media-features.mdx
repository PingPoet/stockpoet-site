import HtmlHead from "./htmlHead.jsx"
import SiteHeader from "./siteHeader.jsx"
import SiteFooter from "./siteFooter.jsx"
import Card from "../../components/card.jsx"
import MediaFeatureCard from "../../components/MediaFeatureCard.jsx"
import { useRemarkSync } from "react-remark"

export const Features = ({ features, site }) => {
  /* we showcase the first feature, and just loop through the rest as cards */
  const showcased =
    features && features.length > 0
      ? features[0]
      : { title: "", description: "" }
  const showcasedTitle = useRemarkSync(showcased.title)
  const showcasedDescription = useRemarkSync(showcased.description)

const others = features && features.length > 0 ? features.slice(1) : []

return (

  <>
    <MediaFeatureCard
      titleMarkdown={showcased.title}
      descriptionMarkdown={showcased.description}
      primaryAction={showcased.primaryAction}
      secondaryAction={showcased.secondaryAction}
      imageUrl={showcased.imageUrl}
      svgUrl={showcased.svgUrl}
      svgFillColor={showcased.svgFillColor}
      svgViewBox={showcased.svgViewBox}
      videoUrl={showcased.videoUrl}
      site={site}
    />

    <div className="py-5">&nbsp;</div>

    {others.map((feature, index) => (
      <MediaFeatureCard
        key={feature.title}
        titleMarkdown={feature.title}
        descriptionMarkdown={feature.description}
        mediaFirst={index % 2 === 0}
        imageUrl={feature.imageUrl}
        svgUrl={feature.svgUrl}
        svgFillColor={feature.svgFillColor}
        svgViewBox={feature.svgViewBox}
        videoUrl={feature.videoUrl}
        site={site}
      />
    ))}

  </>

)
}

<HtmlHead {...props} />

  <body>
    <div className="col-lg-8 mx-auto p-4 py-md-5">

      <SiteHeader {...props} />

      <main className="container-sm" style={{maxWidth:"1024px"}}>

        <Features {...props}/>

      </main>

      <SiteFooter {...props} />

    </div>

  </body>
